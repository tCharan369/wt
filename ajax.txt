BookServlet.java

package com.vnr.servlet;

import javax.servlet.*;
import javax.servlet.http.*;
import java.io.*;
import javax.xml.parsers.*;
import org.w3c.dom.*;

public class BookServlet extends HttpServlet {

    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws IOException {

        response.setContentType("application/json");
        PrintWriter out = response.getWriter();

        try {
            InputStream is = getClass().getClassLoader().getResourceAsStream("books.xml");

            DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
            DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
            Document doc = dBuilder.parse(is);

            NodeList list = doc.getElementsByTagName("book");

            StringBuilder json = new StringBuilder("[");
            
            for (int i = 0; i < list.getLength(); i++) {

                Element e = (Element) list.item(i);

                String title = e.getElementsByTagName("title").item(0).getTextContent();
                String author = e.getElementsByTagName("author").item(0).getTextContent();
                String price = e.getElementsByTagName("price").item(0).getTextContent();

                json.append("{\"title\":\"").append(title)
                    .append("\",\"author\":\"").append(author)
                    .append("\",\"price\":").append(price).append("}");

                if (i != list.getLength() - 1) json.append(",");
            }

            json.append("]");
            out.print(json.toString());

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

function loadBooks() {

    fetch("books")
    .then(res => res.json())
    .then(data => {

        let output = "";

        data.forEach(book => {
            output += `
                <div class="book">
                    <h3>${book.title}</h3>
                    <p>Author: ${book.author}</p>
                    <p>Price: в‚№${book.price}</p>
                </div>
            `;
        });

        document.getElementById("book-list").innerHTML = output;
    });
}
